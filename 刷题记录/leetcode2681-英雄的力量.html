func sumOfPower(nums []int) (ans int) {
    const P int = 1e9 + 7
    sort.Ints(nums)
    pr := 0
    for _, v := range nums {
        res := v * pr % P * v
        ans = (ans + res) % P
        ans = (ans + v*v%P*v) % P
        pr = (2*pr + v) % P
    }
    return ans
}
